10.2.3.1 周年判定の詳細ロジック

function isUserAnniversary(user, today) {
  if (!user.registeredAt) return false;
  const reg = new Date(user.registeredAt);
  const t = new Date(today);
  // 同じ月日かつ少なくとも1年以上経過
  return reg.getMonth() === t.getMonth() &&
         reg.getDate() === t.getDate() &&
         t.getFullYear() > reg.getFullYear();
}

function getYearsCount(user, today) {
  return new Date(today).getFullYear() - new Date(user.registeredAt).getFullYear();
}

「N YEAR(S)」の N は getYearsCount から取得。1年なら "1 YEAR"、2年以上なら "N YEARS"。

§10.4 抽選アルゴリズム

10.4.1 便箋枠の抽選 (詳細)

function pickFrame(track, user, today, history) {
  // === Priority 1: 記念日 (強制発動、レア抽選より優先) ===
  if (isUserBirthday(user, today)) return 'O';
  if (isUserAnniversary(user, today)) return 'P';
  
  // === Priority 2: レア抽選 ===
  if (Math.random() < 0.01) return 'Q';  // 1% Gold
  if (Math.random() < 0.05) return 'R';  // 5% Silver (Gold外し条件付き)
  
  // === Priority 3: 通常抽選プール構築 ===
  let pool = [...ARTIST_FRAMES[track.artist]];
  pool = pool.concat(activeSeasonalFrames(today));
  
  // === Priority 4: 直前と違う枠の保証 ===
  const prev = history.lastFrameOf(user.id, track.id);
  if (prev && pool.includes(prev) && pool.length > 1) {
    pool = pool.filter(f => f !== prev);
  }
  
  // === 均等抽選 ===
  return pool[Math.floor(Math.random() * pool.length)];
}

function activeSeasonalFrames(today) {
  const m = new Date(today).getMonth() + 1;
  const result = [];
  if (m >= 3 && m <= 4) result.push('K');   // 桃
  if (m === 12 || m === 1) result.push('L'); // 雪結晶
  if (m >= 7 && m <= 8) result.push('M');    // 海・波
  if (m >= 10 && m <= 11) result.push('N');  // 紅葉
  return result;
}

function isUserBirthday(user, today) {
  if (!user.birthday) return false;
  const b = new Date(user.birthday);
  const t = new Date(today);
  return b.getMonth() === t.getMonth() && b.getDate() === t.getDate();
}

10.4.1.1 「直前と違う枠」判定の詳細

// history.lastFrameOf の実装
function lastFrameOf(userId, trackId) {
  const history = user.letterHistory || [];
  // 同じトラックで送信した最新のレターの frameId を取得
  const last = history
    .filter(h => h.trackId === trackId)
    .sort((a, b) => b.sentDate.localeCompare(a.sentDate))[0];
  return last ? last.frameId : null;
}
意図: 同じユーザーが同じ曲を再送信した時、直前と同じ枠は除外する。ただし、プールが1枠しかない場合 (例: gEN の枠が全部排他された) は除外せず、同じ枠で再送信される。

10.4.1.2 抽選ロジックの境界条件

状況挙動
誕生日 + ゴールド条件同時成立誕生日枠O が優先 (Priority 1 > 2)
周年 + 季節枠周年枠P が優先 (記念日は強制)
同じ曲を2回連続送信、プール1枠のみ同じ枠で送信される (除外しない)
レア未当選、季節枠+アーティスト枠合計0発生しない (アーティスト枠は最低5枠ある)
letterHistory が空 (初回送信)prev=null、除外なしで全プールから抽選

10.4.2 closing 一文の抽選 (誕生日月特例あり)

function pickClosing(track, user, today, frame) {
  const nick = user.nickname || 'listener';
  
  // === Priority 1: 誕生日枠なら専用 closing ===
  if (frame === 'O') {
    return random(BIRTHDAY_CLOSINGS).replaceAll('●●', nick);
  }
  
  // === Priority 2: 周年枠なら専用 closing ===
  if (frame === 'P') {
    return random(ANNIVERSARY_CLOSINGS).replaceAll('●●', nick);
  }
  
  // === Priority 3: 誕生月の50%判定 ===
  // 「誕生月で、かつ枠が誕生日枠じゃない」場合に発動
  if (isUserBirthMonth(user, today) && Math.random() < 0.5) {
    return random(BIRTHMONTH_CLOSINGS).replaceAll('●●', nick);
  }
  
  // === Priority 4: 通常抽選 (共通+アーティスト別+曲別) ===
  const pool = [
    ...COMMON_CLOSINGS,
    ...ARTIST_CLOSINGS[track.artist],
    ...(track.closings || [])
  ];
  return random(pool).replaceAll('●●', nick);
}

function isUserBirthMonth(user, today) {
  if (!user.birthday) return false;
  return new Date(user.birthday).getMonth() === new Date(today).getMonth();
}

10.7.1 保護5層の詳細

層仕様適用範囲

10.7.4 メール版での保護の限界

メール本文は受信側のクライアントで開かれるため、JavaScript による保護は効かない。Webサイト版でのみ5層保護を適用し、メール版はウォーターマークと「無断転載禁止」フッタのみで対応する。これは仕様上の制約として明記する。

§10.9 レター送信フロー

10.9.1 トリガー条件

  1. ユーザーがログインしている (Cognito 認証済み)
  2. 該当曲がフル試聴アンロック済み (coin消費による解除済み、§9参照)
  3. 該当曲のレターを過去に送信していても、再送信可能 (制限なし、ただし枠は直前と違うものが選ばれる)
coin消費との関係: レター受信は追加coinなし。フル試聴アンロックの coin消費に含まれている (おまけ機能扱い)。これは将来「レターのみ」を追加coin で受け取る選択肢を §22 に登録するか検討。

10.9.3 抽選実行とサーバ保存

// クライアント or サーバで実行
async function sendLetter(trackId, user) {
  const today = new Date().toISOString();
  const track = getTrack(trackId);
  const history = JSON.parse(user.letterHistory || '[]');
  
  // 抽選
  const frame = pickFrame(track, user, today, { lastFrameOf: (u,t) => 
    history.filter(h => h.trackId === t).sort((a,b) => 
      b.sentDate.localeCompare(a.sentDate))[0]?.frameId 
  });
  const closing = pickClosing(track, user, today, frame);
  
  // メール本文HTML生成
  const html = renderLetterHTML({ track, user, frame, closing, sentDate: today });
  
  // メール送信
  await fetch('https://tamjump-member-api.animalb001.workers.dev/send-letter', {
    method: 'POST',
    headers: { 'Authorization': `Bearer ${user.idToken}` },
    body: JSON.stringify({ to: user.email, html, subject: `${track.title} from ${track.artist}` })
  });
  
  // 履歴更新 (Cognito custom属性へ書き戻し)
  history.push({ trackId, frameId: frame, sentDate: today, rare: ['Q','R'].includes(frame) });
  if (history.length > 10) history.shift(); // FIFO
  await updateUserAttribute('custom:letterHistory', JSON.stringify(history));
}

10.9.4 メール送信 (Resend API、§タスク②で確定後)

Cloudflare Workers (tamjump-member-api) 経由で Resend API を叩く。直接呼び出すとAPIキーがクライアント露出するため、Workers をプロキシとして使う。

// Workers エンドポイント擬似コード
export default {
  async fetch(req, env) {
    if (req.url.endsWith('/send-letter')) {
      const auth = req.headers.get('Authorization');
      const user = await verifyJWT(auth, env.COGNITO_JWKS);
      if (!user) return new Response('Unauthorized', { status: 401 });
      
      const { to, html, subject } = await req.json();
      // 送信先がユーザー本人のメールと一致するか確認
      if (to !== user.email) return new Response('Forbidden', { status: 403 });
      
      const resp = await fetch('https://api.resend.com/emails', {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${env.RESEND_API_KEY}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          from: 'TAMSIC ',
          to, subject, html
        })
      });
      
      return new Response(JSON.stringify({ success: resp.ok }), {
        headers: { 'Content-Type': 'application/json' }
      });
    }
  }
};

10.9.6 失敗時の挙動

失敗シナリオ挙動
Workers が 5xx エラー「送信に失敗しました。時間を置いて再度お試しください」表示。coin は消費しない (フル試聴の coin は元から消費済みなので、追加コストなし)
Workers が 401/403ログイン期限切れ → ログイン画面にリダイレクト
Resend が 4xx エラー (アドレス不正)「メールアドレスを確認してください」表示。アドレス変更を促す
letterHistory 書き込み失敗サイレント (メール送信はすでに成功済み、履歴は次回正常時に同期)
抽選ロジック例外 (プール空)フォールバック: 枠 'A'、closing は COMMON_CLOSINGS[0]

§10.10 メール本文HTML生成

10.10.1 制約
